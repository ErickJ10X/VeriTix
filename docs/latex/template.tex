\newcommand{\makefoccover}{%
  \begin{titlepage}
    \thispagestyle{empty}
    \newgeometry{margin=0pt}

    \begin{tikzpicture}[remember picture, overlay]
      % ── Fondo blanco completo ──
      \fill[white] (current page.south west) rectangle (current page.north east);

      % ── Barra decorativa superior (azul institucional) ──
      \fill[focblue]
        (current page.north west) rectangle
        ([yshift=-2.8cm] current page.north east);

      % ── Marca institucional — logo + texto dentro de la barra azul ──
      $if(logo-path)$
      \node[anchor=north west, inner sep=0pt, xshift=0.7cm, yshift=-0.15cm]
        at (current page.north west) {
          \includegraphics[height=3.8cm]{$logo-path$}
      };

      \node[anchor=north west, inner sep=0pt, xshift=5.5cm, yshift=-0.15cm]
        at (current page.north west) {
          \parbox[c][3.5cm][c]{10cm}{%
            \color{white}\bfseries\sffamily\fontsize{15}{15}\selectfont \textbf{\uppercase{$institution-name$}}
          }
      };
      $else$
      \node[anchor=north west, inner sep=0pt, xshift=0.95cm, yshift=-0.08cm]
        at (current page.north west) {
          \parbox[c][2.0cm][c]{14cm}{%
            \color{white}\bfseries\sffamily\fontsize{15}{15}\selectfont \textbf{\uppercase{$institution-name$}}
          }
      };
      $endif$

      % ── Línea divisoria sutil ──
      \draw[focblue, line width=0.4pt]
        ([yshift=-2.8cm] current page.north west) --
        ([yshift=-2.8cm] current page.north east);

      % ── Título "VeriTix" — centrado, parte superior ──
      \node[anchor=center, yshift=3.5cm]
        at (current page.center) {
          \color{focblue}\fontsize{72}{74}\selectfont\bfseries\sffamily $title$
      };

      % ── Subtítulo — debajo del título ──
      \node[anchor=center, inner sep=0pt, yshift=1.2cm]
        at (current page.center) {
          \color{focmid}\fontsize{17}{20}\selectfont\sffamily
          $subtitle$
      };
